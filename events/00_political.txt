# 공산주의의 대두
country_event = { 
	id = ioipoliticalcom.1
	title = ioipoliticalcom.1.t
	desc = ioipoliticalcom.1.d
	picture = GFX_report_event_worker_protests

	trigger = {
		is_democratic = no
		is_communism = no
		communism > 0.2
		NOT = { communism > 0.49 }
		has_unique_national_focus = no
	}
	fire_only_once = yes	
	mean_time_to_happen = { days = 30 }

	# 정부에서 그런 생각을 하는 사람이 있을수도 있겠지
	option = {
		name = ioipoliticalcom.1.a
		custom_effect_tooltip = ioipoliticalcom.1.a.tt
		hidden_effect = {
			set_country_flag = communism_political_movement
		}
	}

	# 인민을 위한 운동은 인민의 지지를 얻어야 하지
	option = {
		name = ioipoliticalcom.1.b
		custom_effect_tooltip = ioipoliticalcom.1.b.tt
		hidden_effect = {
			set_country_flag = communism_popular_movement
		}
	}
}

country_event = {
	id = ioipoliticalcom.2
	title = ioipoliticalcom.2.t
	desc = ioipoliticalcom.2.d
	picture = GFX_report_event_worker_protests

	trigger = {
		is_democratic = no
		is_communism = no
		communism > 0.5
		has_unique_national_focus = no
		has_country_flag = communism_popular_movement
	}
	
	mean_time_to_happen = {
		days = 30
		modifier = {
			factor = 0.5
			has_country_flag = communism_popular_movement

		hidden_effect = {
			clr_country_flag = communism_popular_movement
			set_country_flag = communism_political_movement
		}
	}
	
	# 국민들 의견을 물어볼까...
	option = {
		name = ioipoliticalcom.2.b
		custom_effect_tooltip = ioipoliticalcom.2.b.tt
		hidden_effect = {
			clr_country_flag = communism_popular_movement
			set_country_flag = communism_government_support
		}
	}
}

country_event = {
	id = ioipoliticalcom.6
	title = ioipoliticalcom.6.t
	desc = ioipoliticalcom.6.d
	picture = GFX_report_event_gathering_protest
	
	trigger = {
		has_country_flag = communism_civilwar
		is_communism = no
		is_democratic = no
		communism > 0.7
		has_unique_national_focus = no
	}

	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 1
			has_country_flag = communism_civilwar
		}
	}
	
	option = {
		name = ioipoliticalcom.6.a
		if = {
			limit = { has_government = fascism }
			start_civil_war = {
				ruling_party = communism
				ideology = fascism
				size = 0.6
			}
		}
		if = {
			limit = { has_government = militarytotalism }
			start_civil_war = {
				ruling_party = communism
				ideology = militarytotalism
				size = 0.6
			}
		}
		if = {
			limit = { has_government = authoritariandemocracy }
			start_civil_war = {
				ruling_party = communism
				ideology = authoritariandemocracy
				size = 0.6
			}
		}
		if = {
			limit = { has_government = autocracy }
			start_civil_war = {
				ruling_party = communism
				ideology = autocracy
				size = 0.6
			}
		}
		swap_government_ideas_communism = yes
		#add_timed_idea = { idea = reign_of_terror days = 730 }
	}
}

country_event = {
	id = ioipoliticalcom.7
	title = ioipoliticalcom.7.t

	picture = GFX_report_event_gathering_protest
	
	trigger = {
		is_communism = no
		is_democratic = yes
		communism > 0.49
		has_unique_national_focus = no
	}

	mean_time_to_happen = {
		days = 150
		modifier = {
			factor = 1
			communism > 0.49
		}
		modifier = {
			factor = 2
			has_country_flag = communism_civilwar_prepare
		}
	}
	
	#국민들의 뜻이 그렇다면야...
	option = {

			elections_allowed = no
		}
		swap_government_ideas_communism = yes
	}

	#논의가 더 필요하다!
	option = {
		name = ioipoliticalcom.7.b
		add_political_power = -100
		add_stability = -0.10
		custom_effect_tooltip = ioipoliticalcom.1.a.tt
		hidden_effect = {
			if = { 
				limit = { has_country_flag = communism_civilwar_prepare }
				clr_country_flag = communism_civilwar_prepare
				set_country_flag = communism_civilwar
				else = { set_country_flag = communism_civilwar_prepare }
			}
		}
	}
}

		OR = {

		}

		#has_idea = fascist_revolutionaries
		is_fascism = no
		has_civil_war = no
	}

		days = 2
	}

		hidden_effect = { remove_ideas = fascist_revolutionaries }

		name = ioipolitical.21.a

			base = 0
		}
		if = {
			limit = { fascism > 0.7 }
			start_civil_war = {
				ideology = fascism
				size = 0.5
			}
			else = {
				start_civil_war = {
					ideology = militarytotalism
					size = 0.5
				}
			}
		}
	}

	option = {
		name = ioipolitical.21.b
		ai_chance = {
			base = 0
		}
		if = {
			limit = { fascism > 0.7 }

				start_civil_war = {
					ruling_party = militarytotalism
					ideology = ROOT
					size = 0.5

				}

	# We need a referendum

				}
			}
		}
	}
}
