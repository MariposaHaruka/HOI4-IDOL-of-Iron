# Hindenburg Trigger (HIDDEN)
news_event = {
	id = global_news.51
	hidden = yes
	
	fire_only_once = yes
	
	trigger = {
		date > 1937.5.5
		date < 1939.5.5
		country_exists = ICG
		country_exists = PRS
		ICG = { has_war = no }
		PRS = { has_war = no }
		PRS = { has_civil_war = no }
		NOT = { has_global_flag = hindenburg_destroyed }
		NOT = { has_global_flag = hindenburg_event_in_progress }
	}

	mean_time_to_happen = {
		days = 240
	}
	
	immediate = {
		if = {
			limit = { has_global_flag = PRS_air_safety_regulations_flag }
			ICG = { news_event = { id = global_news.43 } }
			else = {
				random_list = {
					70 = { ICG = { news_event = { id = global_news.42 } } }
					30 = { ICG = { news_event = { id = global_news.43 } } }
				}
			}
		}
	}
	option = {
		name = global_news.51.a
	}
}

news_event = {
	id = ioi_news.1002
	title = ioi_news.1002.t
	desc = ioi_news.1002.d
	picture = GFX_news_event_discover_of_nuclear_fission

	
	major = yes
	
	is_triggered_only = yes

	immediate = {
		set_global_flag = discovery_of_nuclear_fission
	}
	
	option = { #그래서 이게 세계 정세랑 무슨 상관이 있지?
		name = ioi_news.1002.a
		trigger = {
			is_major = no
			NOT = { 
				original_tag = BLA 
				original_tag = LAS
				original_tag = IML
				original_tag = ICG
			}
		}
	}
	option = { #그래서 이게 세계 정세랑 무슨 상관이 있지?
		name = ioi_news.1002.b
		trigger = { 
			OR = { 
				is_major = yes
				original_tag = IML
				original_tag = ICG
			}
			NOT = { original_tag = BLA } # 얘네는 밑 쪽으로 뜸
		}
		custom_effect_tooltip = discovery_of_nuclear_fission_tt
	}
	option = { #음...
		name = ioi_news.1002.c

			OR = { 
				original_tag = BLA

			}
		}

	}
}
