	ignite_the_fascist_civil_war = {
		icon = generic_ignite_civil_war
		available = {
			has_stability < 0.5
		}
		visible = {
			has_country_flag = preparation_for_fascist_civil_war
			NOT = { has_government = fascism }
			has_unique_national_focus = no
			check_variable = { num_owned_states > 1 }
		}
		fire_only_once = yes
		cost = 50
		complete_effect = {
			if = {
				limit = { is_democratic = yes }
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@conservatism
				}
				add_to_variable = {
					var = civil_war_size_var
					value = party_popularity@liberalism_mod
				}
				add_to_variable = {
					var = civil_war_size_var
					value = party_popularity@progressivism
				}
				add_to_variable = {
					var = civil_war_size_var
					value = party_popularity@socialdemocracy
				}
			}
			if = {
				limit = { has_government = communism }
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@communism
				}
			}
			if = {
				limit = { has_government = authoritariancommunism }
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@authoritariancommunism
				}
			}
			if = {
				limit = { has_government = socialism }
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@socialism
				}
			}
			if = {
				limit = { has_government = anarchosyndicalism }
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@anarchosyndicalism
				}
			}
			if = {
				limit = { has_government = authoritariandemocracy }
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@authoritariandemocracy
				}
			}
			if = {
				limit = { has_government = autocracy }
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@autocracy
				}
			}
			if = {
				limit = { has_government = militarytotalism }
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@militarytotalism
				}
			}
			subtract_from_variable = {
				var = civil_war_size_var
				value = army_support_var
			}
			if = {
				limit = {
					check_variable = {
						var = civil_war_size_var
						value = 0.3
						compare = less_than
					}
				}
				set_variable = {
					var = civil_war_size_var
					value = 0.3
				}
			}
			start_civil_war = {
				ruling_party = fascism
				ideology = ROOT
				size = civil_war_size_var
				keep_unit_leaders_trigger = {
					has_trait = hidden_sympathies
				}
			}
			clr_country_flag = preparation_for_fascism_civil_war
			clr_country_flag = military_support_for_fascism_civil_war
			clr_country_flag = civil_support_for_fascism_civil_war
			set_country_flag = ideology_civil_war
		}
		ai_will_do = {
			factor = 0
		}
	}
