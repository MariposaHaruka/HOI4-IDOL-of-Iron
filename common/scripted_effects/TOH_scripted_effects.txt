TOH_paranoia_increase_based_on_unit_leader_level_effect = {
	if = {
		limit = {
			has_country_flag = TOH_paranoia_system_active_flag
		}
		#custom_effect_tooltip = TOH_paranoia_increase_based_on_unit_leader_level_effect_tt
		add_to_variable = { TOH_paranoia = TOH_unit_leader_level }
		TOH_paranoia_clamp_and_update_ui_effect = yes
		clear_variable = TOH_unit_leader_level
	}
}

TOH_paranoia_decrease_based_on_unit_leader_level_effect = {
	if = {
		limit = {
			has_country_flag = TOH_paranoia_system_active_flag
		}
		#custom_effect_tooltip = TOH_paranoia_decrease_based_on_unit_leader_level_effect_tt
		subtract_from_variable = { TOH_paranoia = TOH_unit_leader_level }
		TOH_paranoia_clamp_and_update_ui_effect = yes
		clear_variable = TOH_unit_leader_level
	}
}

TOH_paranoia_modifier_increase_effect = { #Called on_weekly to apply the modifier
	custom_effect_tooltip = TOH_paranoia_modifier_increase_effect_tt
	set_temp_variable = { TOH_paranoia_total_weekly_modifier_increase = TOH_paranoia_modifier_increase } #Get default weekly modifier increase value
	multiply_temp_variable = { TOH_paranoia_total_weekly_modifier_increase = TOH_paranoia_weekly_modifiers_amount } #Multiply default weekly modifier increase by the number of active weekly modifiers
	add_to_variable = { TOH_paranoia = TOH_paranoia_total_weekly_modifier_increase } #Apply the result and add to Paranoia
	TOH_paranoia_clamp_and_update_ui_effect = yes
}

TOH_paranoia_modifier_update_gui_tooltip = { #Only updates the tooltip in the decision GUI
	ROOT = { set_variable = { TOH_paranoia_weekly_modifier_ui = TOH_paranoia_modifier_increase } } #Get default weekly modifier increase value
	ROOT = { multiply_variable = { TOH_paranoia_weekly_modifier_ui = TOH_paranoia_weekly_modifiers_amount } } #Multiply default weekly modifier increase by the number of active weekly modifiers
}

TOH_paranoia_clamp_and_update_ui_effect = {
	clamp_variable = {
		var = TOH_paranoia
		min = 0
		max = 100
	}
	TOH_update_ui_effect = yes
}

### 5개년 계획 중지 ###
TOH_remove_5year_plan = {
	if = {
		limit = { has_idea = TOH_2nd_5_year_plan }
		remove_ideas = TOH_2nd_5_year_plan
	}
	else_if = {
		limit = { has_idea = TOH_3rd_5_year_plan }
		remove_ideas = TOH_3rd_5_year_plan
	}
	else_if = {
		limit = { has_idea = TOH_4th_5_year_plan }
		remove_ideas = TOH_4th_5_year_plan
	}
}
