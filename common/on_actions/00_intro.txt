on_actions = {
    on_startup = {
	    effect = {
			every_country = {
				limit = { is_ai = no }
				add_to_array = { # It will automatically add the scope
					array = global.players
				}
			}
			for_each_scope_loop = {
				array = global.players
				if = {
					limit = { tag_has_country_intro = yes }
					set_variable = { intro_variable = 1 }
				}
				else = {
					set_variable = { intro_variable = 4 }
				}
			}
        }
    }
}
