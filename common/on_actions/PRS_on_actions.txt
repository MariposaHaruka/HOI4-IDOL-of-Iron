on_actions = {
	#시작 국민정신 설정
	on_war = {
		effect = {
			remove_ideas = PRS_rearmament_plan
		}
	}
	on_state_control_changed = {
		effect = { #Check how many occupied states Germany has
			if = {
				limit = {
					ROOT = { original_tag = PRS }
					FROM.FROM = {
						NOT = {	is_core_of = ROOT }
						is_on_continent = europe
					}
				}
				add_to_variable = { PRS_occupied_states_europe = 1 }
			}
			if = {
				limit = {
					FROM = { original_tag = PRS }
					FROM.FROM = {
						NOT = {	is_core_of = FROM }
						is_on_continent = europe
					}
				}
				subtract_from_variable = { PRS_occupied_states_europe = 1 }
			}
		}
	}

	on_capitulation_immediate = {
		effect = {
			if = {
				limit = {
					FROM = { original_tag = PRS }
					ROOT = { is_major = yes }
				}
				FROM = { add_to_variable = { GER_capitulated_major_countries = 1 } }
			}
		}
	}

	on_capitulation = {
		effect = {
			if = {
				limit = {
					FROM = {
						OR = {
							original_tag = PRS
							is_ally_with = PRS
						}
					}
					ROOT = {
						OR = {
							capital_scope = { is_on_continent = europe }
							original_tag = GUP
						}
					}
				}
				ROOT = {
					set_country_flag = PRS_has_capitulated_to_projectsekai
					add_to_array = { 
						array = PRS.PRS_seize_gold_targets_array 
						value = THIS
					}
				}
			}
		}
	}
}
